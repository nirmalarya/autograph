## 1. ThemeProvider Enhancement (UPDATED)

### File: `services/frontend/app/components/ThemeProvider.tsx`

**Purpose:**
- Extended to support high contrast mode
- Manages high contrast state alongside theme state
- Persists preference to localStorage
- Applies high-contrast class to document root

**Implementation:**
```typescript
interface ThemeContextType {
  theme: Theme;
  setTheme: (theme: Theme) => void;
  resolvedTheme: 'light' | 'dark';
  highContrast: boolean;           // NEW
  setHighContrast: (enabled: boolean) => void;  // NEW
}

export function ThemeProvider({ children }: { children: React.ReactNode }) {
  const [theme, setThemeState] = useState<Theme>('system');
  const [resolvedTheme, setResolvedTheme] = useState<'light' | 'dark'>('light');
  const [highContrast, setHighContrastState] = useState<boolean>(false);  // NEW

  // Initialize from localStorage
  useEffect(() => {
    // ... theme initialization
    
    // NEW: Initialize high contrast mode
    const storedHighContrast = localStorage.getItem('highContrast');
    if (storedHighContrast === 'true') {
      setHighContrastState(true);
    }
  }, []);

  // Apply high contrast class
  useEffect(() => {
    const updateResolvedTheme = () => {
      // ... theme logic
      
      // NEW: Apply high contrast mode
      if (highContrast) {
        root.classList.add('high-contrast');
      } else {
        root.classList.remove('high-contrast');
      }
      
      root.setAttribute('data-high-contrast', highContrast.toString());
    };
    // ...
  }, [theme, highContrast]);  // NEW: Added highContrast dependency

  const setHighContrast = (enabled: boolean) => {
    setHighContrastState(enabled);
    localStorage.setItem('highContrast', enabled.toString());
  };

  return (
    <ThemeContext.Provider value={{ 
      theme, setTheme, resolvedTheme, 
      highContrast, setHighContrast  // NEW
    }}>
      {children}
    </ThemeContext.Provider>
  );
}
```

**Features:**
- High contrast state management
- localStorage persistence
- Automatic class application
- Works with both light and dark themes
- Data attribute for CSS targeting

```typescript
'use client';

import { useTheme } from './ThemeProvider';

export default function HighContrastToggle() {
  const { highContrast, setHighContrast } = useTheme();

  const toggleHighContrast = () => {
    setHighContrast(!highContrast);
  };

  return (
    <button
      onClick={toggleHighContrast}
      className={`p-2 rounded-lg transition-colors ${
        highContrast
          ? 'bg-blue-600 text-white hover:bg-blue-700'
          : 'hover:bg-gray-200 dark:hover:bg-gray-700'
      }`}
      title={`High Contrast: ${highContrast ? 'On' : 'Off'}`}
      aria-label="Toggle high contrast mode"
      aria-pressed={highContrast}
    >
      {/* Eye icon when enabled, contrast icon when disabled */}
      <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
        {highContrast ? (
          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2}
            d="M15 12a3 3 0 11-6 0 3 3 0 016 0z M2.458 12C3.732 7.943 7.523 5 12 5c4.478 0 8.268 2.943 9.542 7-1.274 4.057-5.064 7-9.542 7-4.477 0-8.268-2.943-9.542-7z"
          />
        ) : (
          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2}
            d="M12 3v1m0 16v1m9-9h-1M4 12H3m15.364 6.364l-.707-.707M6.343 6.343l-.707-.707m12.728 0l-.707.707M6.343 17.657l-.707.707M16 12a4 4 0 11-8 0 4 4 0 018 0z"
          />
        )}
      </svg>
      {highContrast && (
        <span className="ml-1 text-xs font-semibold">HC</span>
      )}
    </button>
  );
}
```

**Features:**
- Uses ThemeProvider hook
- Visual indicator (blue background) when enabled
- "HC" badge when active
- Accessible ARIA labels

- Tooltip shows current state
