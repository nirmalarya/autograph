**Objective:** Different shortcuts on different pages (canvas vs dashboard vs note)

### Existing Implementation Verified:

**Dashboard Page:**
```typescript
// services/frontend/app/dashboard/page.tsx
useEffect(() => {
  const handleKeyDown = (e: KeyboardEvent) => {
    // Cmd+N or Ctrl+N - Create new diagram
    if ((e.metaKey || e.ctrlKey) && e.key === 'n') {
      e.preventDefault();
      setShowCreateModal(true);
    }
    
    // Cmd+F or Ctrl+F - Focus search
    if ((e.metaKey || e.ctrlKey) && e.key === 'f') {
      e.preventDefault();
      searchInputRef.current?.focus();
    }
    
    // Cmd+K or Ctrl+K - Command palette
    if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
      e.preventDefault();
      setShowCommandPalette(true);
    }
    
    // Cmd+? or Ctrl+? - Keyboard shortcuts
    if ((e.metaKey || e.ctrlKey) && e.shiftKey && e.key === '/') {
      e.preventDefault();
      setShowKeyboardShortcuts(true);
    }
  };
  
  window.addEventListener('keydown', handleKeyDown);
  return () => window.removeEventListener('keydown', handleKeyDown);
}, []);
```

**Canvas Page:**
```typescript
// services/frontend/app/canvas/[id]/page.tsx
useEffect(() => {
  const handleKeyDown = (event: KeyboardEvent) => {
    // Ctrl+S (Windows/Linux) or Cmd+S (Mac)
    if ((event.ctrlKey || event.metaKey) && event.key === 's') {
      event.preventDefault();
      
      const editor = (window as any).tldrawEditor;
      if (editor && !saving) {
        handleSave(editor);
      }
    }
  };
  
  window.addEventListener('keydown', handleKeyDown);
  return () => window.removeEventListener('keydown', handleKeyDown);
}, [saving]);
```
