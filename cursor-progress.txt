const supportsWebP = useRef<boolean | null>(null);

useEffect(() => {
  if (supportsWebP.current === null) {
    const checkWebPSupport = () => {
      const canvas = document.createElement('canvas');
      if (canvas.getContext && canvas.getContext('2d')) {
        return canvas.toDataURL('image/webp').indexOf('data:image/webp') === 0;
      }
      return false;
    };
    supportsWebP.current = checkWebPSupport();
  }

  // Convert image URL to WebP if supported
  if (src && supportsWebP.current && !src.endsWith('.webp')) {
    const webpSrc = src.replace(/\.(png|jpg|jpeg)$/i, '.webp');
    setImageSrc(webpSrc);
  } else {
    setImageSrc(src);
  }
}, [src]);

// Preload critical images
export function usePreloadImage(src: string) {
  useEffect(() => {
    if (!src) return;
    const link = document.createElement('link');
    link.rel = 'preload';
    link.as = 'image';
    link.href = src;
    document.head.appendChild(link);
    return () => {
      document.head.removeChild(link);
    };
  }, [src]);
}

// Convert image URL to WebP
export function getWebPUrl(url: string): string {

  if (url.endsWith('.webp')) return url;

}

## 1. WebP Format Implementation

**Why WebP:**
- ‚úÖ 25-35% smaller file sizes than PNG/JPEG
- ‚úÖ Better compression with same quality
- ‚úÖ Supported by all modern browsers (95%+ coverage)
- ‚úÖ Faster page loads and less bandwidth
- ‚úÖ Better user experience

**Why Automatic Fallback:**
- ‚úÖ Graceful degradation for older browsers
- ‚úÖ No manual intervention needed
- ‚úÖ Tries WebP first, falls back to original
- ‚úÖ Transparent to users
- ‚úÖ Works with any image format

**Why Client-Side Detection:**
- ‚úÖ Browser-specific support detection
- ‚úÖ No server-side logic needed
- ‚úÖ Works with any image source
- ‚úÖ Cached for performance
- ‚úÖ Reliable and tested

## 2. Lazy Loading Implementation

- ‚úÖ Native browser API (performant)
- ‚úÖ Better than scroll listeners
- ‚úÖ Automatic cleanup
- ‚úÖ Configurable thresholds
- ‚úÖ Wide browser support

**Why 50px Root Margin:**
- ‚úÖ Start loading before image enters viewport

- ‚úÖ Balances performance and UX
- ‚úÖ Prevents white flashes
