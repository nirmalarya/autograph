      topic.title.toLowerCase().includes(searchQuery.toLowerCase()) ||

#### 12. useHelpCenter Hook
**Location:** `services/frontend/app/components/HelpCenter.tsx`

**Functions:**
```typescript
export function useHelpCenter() {
  const [isOpen, setIsOpen] = useState(false);
  const [defaultTopic, setDefaultTopic] = useState<string | undefined>(undefined);

  const open = useCallback((topic?: string) => {
    setDefaultTopic(topic);
    setIsOpen(true);
  }, []);

  const close = useCallback(() => {
    setIsOpen(false);
    setDefaultTopic(undefined);
  }, []);

  const toggle = useCallback(() => {
    setIsOpen(prev => !prev);
  }, []);

  return {
    isOpen,
    defaultTopic,
    open,
    close,
    toggle,
  };
}
```

**Usage:**
- `open()` - Opens the help center
- `open('topic-id')` - Opens with specific topic
- `close()` - Closes the help center
- `toggle()` - Toggles the help center state
- `isOpen` - Current state
- `defaultTopic` - Topic to open by default

**Keyboard Shortcut:**
- Question mark (?) key
- Works globally (except in inputs/textareas)
- Toggles help center
- Prevents default behavior
- Excludes contentEditable elements

**Implementation:**
```typescript
useEffect(() => {
  const handleKeyDown = (e: KeyboardEvent) => {
    if (e.key === '?' && !e.ctrlKey && !e.metaKey && !e.altKey) {
      const target = e.target as HTMLElement;
      if (
        target.tagName === 'INPUT' ||
        target.tagName === 'TEXTAREA' ||
        target.isContentEditable
      ) {
        return;
      }
      
      e.preventDefault();
      setIsOpen(prev => !prev);
    }
  };

  window.addEventListener('keydown', handleKeyDown);
  return () => window.removeEventListener('keydown', handleKeyDown);
}, [mounted]);
```
